function calculateTrustScore() { let b = log.filter(e => e.k === "Backspace").length; let k = log.length; if (k < 10) return; // 1. Backspace Factor (Humans usually backspace 8-15% of the time) let backspaceRatio = (b / k); let errorScore = backspaceRatio > 0.05 ? 50 : (backspaceRatio * 1000); // 2. Burst Factor (Measures if you type in biological bursts vs consistent AI speed) // We check the timing between the last 10 keys let latencies = []; for(let i = log.length - 10; i < log.length - 1; i++) { if(log[i] && log[i+1]) latencies.push(log[i+1].t - log[i].t); } let variance = 0; if(latencies.length > 0) { let avg = latencies.reduce((a,b) => a+b) / latencies.length; variance = latencies.map(x => Math.pow(x - avg, 2)).reduce((a,b) => a+b) / latencies.length; } // If variance is high (human-like), add up to 50 points let rhythmScore = Math.min(50, variance / 500); let finalScore = Math.floor(errorScore + rhythmScore); document.getElementById('trust-score').innerText = Math.min(100, finalScore) + "%"; }
